82impl PushService {
83    pub fn new(
84        cfg: impl Into<ServiceConfiguration>,
85        credentials: Option<ServiceCredentials>,
86        user_agent: impl AsRef<str>,
87    ) -> Self {
88        let cfg = cfg.into();
89        let client = reqwest::ClientBuilder::new()
90            .tls_built_in_root_certs(false)
91            .add_root_certificate(
92                reqwest::Certificate::from_pem(
93                    cfg.certificate_authority.as_bytes(),
94                )
95                .unwrap(),
96            )
97            .connect_timeout(Duration::from_secs(10))
98            .timeout(Duration::from_secs(65))
99            .user_agent(user_agent.as_ref())
100            .build()
101            .unwrap();
102
103        Self {
104            cfg,
105            credentials: credentials.and_then(|c| c.authorization()),
106            client,
107        }
108    }
109
110    #[tracing::instrument(skip(self), fields(endpoint = %endpoint))]
111    pub fn request(
112        &self,
113        method: Method,
114        endpoint: Endpoint,
115        auth_override: HttpAuthOverride,
116    ) -> Result<RequestBuilder, ServiceError> {
117        let url = endpoint.into_url(&self.cfg)?;
118        let mut builder = self.client.request(method, url);
119
120        builder = match auth_override {
121            HttpAuthOverride::NoOverride => {
122                if let Some(HttpAuth { username, password }) =
123                    self.credentials.as_ref()
124                {
125                    builder.basic_auth(username, Some(password))
126                } else {
127                    builder
128                }
129            },
130            HttpAuthOverride::Identified(HttpAuth { username, password }) => {
131                builder.basic_auth(username, Some(password))
132            },
133            HttpAuthOverride::Unidentified => builder,
134        };
135
136        Ok(builder)
137    }

139    pub async fn ws<C: WebSocketType>(
140        &mut self,
141        path: &str,
142        keepalive_path: &str,
143        additional_headers: &[(&'static str, &str)],
144        credentials: Option<ServiceCredentials>,
145    ) -> Result<SignalWebSocket<C>, ServiceError> {
146        let span = debug_span!("websocket");
147
148        let mut url = Endpoint::service(path).into_url(&self.cfg)?;
149        url.set_scheme("wss").expect("valid https base url");
150
151        let mut builder = self.client.get(url);
152        for (key, value) in additional_headers {
153            builder = builder.header(*key, *value);
154        }
155
156        if let Some(credentials) = credentials {
157            builder =
158                builder.basic_auth(credentials.login(), credentials.password);
159        }
160
161        let ws = builder
162            .upgrade()
163            .send()
164            .await?
165            .into_websocket()
166            .instrument(span.clone())
167            .await?;
168
169        let unidentified_push_service = PushService {
170            cfg: self.cfg.clone(),
171            credentials: None,
172            client: self.client.clone(),
173        };
174        let (ws, task) = SignalWebSocket::new(
175            ws,
176            keepalive_path.to_owned(),
177            unidentified_push_service,
178        );
179        let task = task.instrument(span);
180        tokio::task::spawn(task);
181        Ok(ws)
182    }

209    pub(crate) trait ProtobufRequestBuilderExt
210    where
211        Self: Sized,
212    {
213        /// Set the request payload encoded as protobuf.
214        /// Sets the `Content-Type` header to `application/protobuf`
215        #[allow(dead_code)]
216        fn protobuf<T: Message + Default>(
217            self,
218            value: T,
219        ) -> Result<Self, EncodeError>;
220    }
221
222    #[async_trait::async_trait]
223    pub(crate) trait ProtobufResponseExt {
224        /// Get the response body decoded from Protobuf
225        async fn protobuf<T>(self) -> Result<T, ServiceError>
226        where
227            T: prost::Message + Default;
228    }
229
230    impl ProtobufRequestBuilderExt for RequestBuilder {
231        fn protobuf<T: Message + Default>(
232            self,
233            value: T,
234        ) -> Result<Self, EncodeError> {
235            let mut buf = Vec::new();
236            value.encode(&mut buf)?;
237            let this =
238                self.header(header::CONTENT_TYPE, "application/protobuf");
239            Ok(this.body(buf))
240        }
241    }
242
243    #[async_trait]
244    impl ProtobufResponseExt for Response {
245        async fn protobuf<T>(self) -> Result<T, ServiceError>
246        where
247            T: Message + Default,
248        {
249            let body = self.bytes().await?;
250            let decoded = T::decode(body)?;
251            Ok(decoded)
252        }
253    }
