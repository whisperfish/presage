85    /// Checks the availability of pre-keys, and updates them as necessary.
86    ///
87    /// Parameters are the protocol's `StoreContext`, and the offsets for the next pre-key and
88    /// signed pre-keys.
89    ///
90    /// Equivalent to Java's RefreshPreKeysJob
91    #[allow(clippy::too_many_arguments)]
92    #[tracing::instrument(skip(self, csprng, protocol_store))]
93    pub async fn update_pre_key_bundle<R: Rng + CryptoRng, P: PreKeysStore>(
94        &mut self,
95        protocol_store: &mut P,
96        service_id_kind: ServiceIdKind,
97        use_last_resort_key: bool,
98        csprng: &mut R,
99    ) -> Result<(), ServiceError> {
100        let prekey_status = match self
101            .service
102            .get_pre_key_status(service_id_kind)
103            .instrument(tracing::span!(
104                tracing::Level::DEBUG,
105                "Fetching pre key status"
106            ))
107            .await
108        {
109            Ok(status) => status,
110            Err(ServiceError::Unauthorized) => {
111                tracing::info!("Got Unauthorized when fetching pre-key status. Assuming first installment.");
112                // Additionally, the second PUT request will fail if this really comes down to an
113                // authorization failure.
114                crate::push_service::PreKeyStatus {
115                    count: 0,
116                    pq_count: 0,
117                }
118            },
119            Err(e) => return Err(e),
120        };
121        tracing::trace!("Remaining pre-keys on server: {:?}", prekey_status);
122
123        // XXX We should honestly compare the pre-key count with the number of pre-keys we have
124        // locally. If we have more than the server, we should upload them.
125        // Currently the trait doesn't allow us to do that, so we just upload the batch size and
126        // pray.
127        if prekey_status.count >= PRE_KEY_MINIMUM
128            && prekey_status.pq_count >= PRE_KEY_MINIMUM
129        {
130            if protocol_store.signed_pre_keys_count().await? > 0
131                && protocol_store.kyber_pre_keys_count(true).await? > 0
132            {
133                tracing::debug!("Available keys sufficient");
134                return Ok(());
135            }
136            tracing::info!("Available keys sufficient; forcing refresh.");
137        }
138
139        let identity_key_pair = protocol_store
140            .get_identity_key_pair()
141            .instrument(tracing::trace_span!("get identity key pair"))
142            .await?;
143
144        let last_resort_keys = protocol_store
145            .load_last_resort_kyber_pre_keys()
146            .instrument(tracing::trace_span!("fetch last resort key"))
147            .await?;
148
149        // XXX: Maybe this check should be done in the generate_pre_keys function?
150        let has_last_resort_key = !last_resort_keys.is_empty();
151
152        let (pre_keys, signed_pre_key, pq_pre_keys, pq_last_resort_key) =
153            crate::pre_keys::replenish_pre_keys(
154                protocol_store,
155                csprng,
156                &identity_key_pair,
157                use_last_resort_key && !has_last_resort_key,
158                PRE_KEY_BATCH_SIZE,
159                PRE_KEY_BATCH_SIZE,
160            )
161            .await?;
162
163        let pq_last_resort_key = if has_last_resort_key {
164            if last_resort_keys.len() > 1 {
165                tracing::warn!(
166                    "More than one last resort key found; only uploading first"
167                );
168            }
169            Some(KyberPreKeyEntity::try_from(last_resort_keys[0].clone())?)
170        } else {
171            pq_last_resort_key
172                .map(KyberPreKeyEntity::try_from)
173                .transpose()?
174        };
175
176        let identity_key = *identity_key_pair.identity_key();
177
178        let pre_keys: Vec<_> = pre_keys
179            .into_iter()
180            .map(PreKeyEntity::try_from)
181            .collect::<Result<_, _>>()?;
182        let signed_pre_key = signed_pre_key.try_into()?;
183        let pq_pre_keys: Vec<_> = pq_pre_keys
184            .into_iter()
185            .map(KyberPreKeyEntity::try_from)
186            .collect::<Result<_, _>>()?;
187
188        tracing::info!(
189            "Uploading pre-keys: {} one-time, {} PQ, {} PQ last resort",
190            pre_keys.len(),
191            pq_pre_keys.len(),
192            if pq_last_resort_key.is_some() { 1 } else { 0 }
193        );
194
195        let pre_key_state = PreKeyState {
196            pre_keys,
197            signed_pre_key,
198            identity_key,
199            pq_pre_keys,
200            pq_last_resort_key,
201        };
202
203        self.service
204            .register_pre_keys(service_id_kind, pre_key_state)
205            .instrument(tracing::span!(
206                tracing::Level::DEBUG,
207                "Uploading pre keys"
208            ))
209            .await?;
210
211        Ok(())
212    }

269    /// Link a new device, given a tsurl.
270    ///
271    /// Equivalent of Java's `AccountManager::addDevice()`
272    ///
273    /// When calling this, make sure that UnidentifiedDelivery is disabled, ie., that your
274    /// application does not send any unidentified messages before linking is complete.
275    /// Cfr.:
276    /// - `app/src/main/java/org/thoughtcrime/securesms/migrations/LegacyMigrationJob.java`:250 and;
277    /// - `app/src/main/java/org/thoughtcrime/securesms/DeviceActivity.java`:195
278    ///
279    /// ```java
280    /// TextSecurePreferences.setIsUnidentifiedDeliveryEnabled(context, false);
281    /// ```
282    pub async fn link_device<R: Rng + CryptoRng>(
283        &mut self,
284        csprng: &mut R,
285        url: url::Url,
286        aci_identity_store: &dyn IdentityKeyStore,
287        pni_identity_store: &dyn IdentityKeyStore,
288        credentials: ServiceCredentials,
289        master_key: Option<MasterKey>,
290    ) -> Result<(), ProvisioningError> {
291        let query: HashMap<_, _> = url.query_pairs().collect();
292        let ephemeral_id =
293            query.get("uuid").ok_or(ProvisioningError::MissingUuid)?;
294        let pub_key = query
295            .get("pub_key")
296            .ok_or(ProvisioningError::MissingPublicKey)?;
297        let pub_key = BASE64_RELAXED
298            .decode(&**pub_key)
299            .map_err(|e| ProvisioningError::InvalidPublicKey(e.into()))?;
300        let pub_key = PublicKey::deserialize(&pub_key)
301            .map_err(|e| ProvisioningError::InvalidPublicKey(e.into()))?;
302
303        let aci_identity_key_pair =
304            aci_identity_store.get_identity_key_pair().await?;
305        let pni_identity_key_pair =
306            pni_identity_store.get_identity_key_pair().await?;
307
308        if credentials.aci.is_none() {
309            tracing::warn!("No local ACI set");
310        }
311        if credentials.pni.is_none() {
312            tracing::warn!("No local PNI set");
313        }
314
315        let provisioning_code = self.new_device_provisioning_code().await?;
316
317        let msg = ProvisionMessage {
318            aci: credentials.aci.as_ref().map(|u| u.to_string()),
319            aci_identity_key_public: Some(
320                aci_identity_key_pair.public_key().serialize().into_vec(),
321            ),
322            aci_identity_key_private: Some(
323                aci_identity_key_pair.private_key().serialize(),
324            ),
325            number: Some(credentials.e164()),
326            pni_identity_key_public: Some(
327                pni_identity_key_pair.public_key().serialize().into_vec(),
328            ),
329            pni_identity_key_private: Some(
330                pni_identity_key_pair.private_key().serialize(),
331            ),
332            pni: credentials.pni.as_ref().map(uuid::Uuid::to_string),
333            profile_key: self.profile_key.as_ref().map(|x| x.bytes.to_vec()),
334            // CURRENT is not exposed by prost :(
335            provisioning_version: Some(i32::from(
336                ProvisioningVersion::TabletSupport,
337            ) as _),
338            provisioning_code: Some(provisioning_code),
339            read_receipts: None,
340            user_agent: None,
341            master_key: master_key.map(|x| x.into()),
342        };
343
344        let cipher = ProvisioningCipher::from_public(pub_key);
345
346        let encrypted = cipher.encrypt(csprng, msg)?;
347        self.send_provisioning_message(ephemeral_id, encrypted)
348            .await?;
349        Ok(())
350    }

381    pub async fn register_account<
382        R: Rng + CryptoRng,
383        Aci: PreKeysStore + IdentityKeyStore,
384        Pni: PreKeysStore + IdentityKeyStore,
385    >(
386        &mut self,
387        csprng: &mut R,
388        registration_method: RegistrationMethod<'_>,
389        account_attributes: AccountAttributes,
390        aci_protocol_store: &mut Aci,
391        pni_protocol_store: &mut Pni,
392        skip_device_transfer: bool,
393    ) -> Result<VerifyAccountResponse, ProvisioningError> {
394        let aci_identity_key_pair = aci_protocol_store
395            .get_identity_key_pair()
396            .instrument(tracing::trace_span!("get ACI identity key pair"))
397            .await?;
398        let pni_identity_key_pair = pni_protocol_store
399            .get_identity_key_pair()
400            .instrument(tracing::trace_span!("get PNI identity key pair"))
401            .await?;
402
403        let (
404            _aci_pre_keys,
405            aci_signed_pre_key,
406            _aci_kyber_pre_keys,
407            aci_last_resort_kyber_prekey,
408        ) = crate::pre_keys::replenish_pre_keys(
409            aci_protocol_store,
410            csprng,
411            &aci_identity_key_pair,
412            true,
413            0,
414            0,
415        )
416        .await?;
417
418        let (
419            _pni_pre_keys,
420            pni_signed_pre_key,
421            _pni_kyber_pre_keys,
422            pni_last_resort_kyber_prekey,
423        ) = crate::pre_keys::replenish_pre_keys(
424            pni_protocol_store,
425            csprng,
426            &pni_identity_key_pair,
427            true,
428            0,
429            0,
430        )
431        .await?;
432
433        let aci_identity_key = aci_identity_key_pair.identity_key();
434        let pni_identity_key = pni_identity_key_pair.identity_key();
435
436        let dar = DeviceActivationRequest {
437            aci_signed_pre_key: aci_signed_pre_key.try_into()?,
438            pni_signed_pre_key: pni_signed_pre_key.try_into()?,
439            aci_pq_last_resort_pre_key: aci_last_resort_kyber_prekey
440                .expect("requested last resort prekey")
441                .try_into()?,
442            pni_pq_last_resort_pre_key: pni_last_resort_kyber_prekey
443                .expect("requested last resort prekey")
444                .try_into()?,
445        };
446
447        let result = self
448            .service
449            .submit_registration_request(
450                registration_method,
451                account_attributes,
452                skip_device_transfer,
453                aci_identity_key,
454                pni_identity_key,
455                dar,
456            )
457            .await?;
458
459        Ok(result)
460    }

516    /// Upload a profile
517    ///
518    /// Panics if no `profile_key` was set.
519    ///
520    /// Returns the avatar url path.
521    pub async fn upload_versioned_profile<
522        's,
523        C: std::io::Read + Send + 's,
524        R: Rng + CryptoRng,
525        S: AsRef<str>,
526    >(
527        &mut self,
528        aci: libsignal_protocol::Aci,
529        name: ProfileName<S>,
530        about: Option<String>,
531        about_emoji: Option<String>,
532        avatar: AvatarWrite<&'s mut C>,
533        csprng: &mut R,
534    ) -> Result<Option<String>, ProfileManagerError> {
535        let profile_key =
536            self.profile_key.expect("set profile key in AccountManager");
537        let profile_cipher = ProfileCipher::new(profile_key);
538
539        // Profile encryption
540        let name = profile_cipher.encrypt_name(name.as_ref(), csprng)?;
541        let about = about.unwrap_or_default();
542        let about = profile_cipher.encrypt_about(about, csprng)?;
543        let about_emoji = about_emoji.unwrap_or_default();
544        let about_emoji = profile_cipher.encrypt_emoji(about_emoji, csprng)?;
545
546        // If avatar -> upload
547        if matches!(avatar, AvatarWrite::NewAvatar(_)) {
548            // FIXME ProfileCipherOutputStream.java
549            // It's just AES GCM, but a bit of work to decently implement it with a stream.
550            unimplemented!("Setting avatar requires ProfileCipherStream")
551        }
552
553        let profile_key = profile_cipher.into_inner();
554        let commitment = profile_key.get_commitment(aci);
555        let profile_key_version = profile_key.get_profile_key_version(aci);
556
557        Ok(self
558            .service
559            .write_profile::<C, S>(
560                &profile_key_version,
561                &name,
562                &about,
563                &about_emoji,
564                &commitment,
565                avatar,
566            )
567            .await?)
568    }

644    /// Initialize PNI on linked devices.
645    ///
646    /// Should be called as the primary device to migrate from pre-PNI to PNI.
647    ///
648    /// This is the equivalent of Android's PnpInitializeDevicesJob or iOS' PniHelloWorldManager.
649    #[tracing::instrument(skip(self, aci_protocol_store, pni_protocol_store, sender, local_aci, csprng), fields(local_aci = local_aci.service_id_string()))]
650    pub async fn pnp_initialize_devices<
651        R: Rng + CryptoRng,
652        AciStore: PreKeysStore + SessionStoreExt,
653        PniStore: PreKeysStore,
654        AciOrPni: ProtocolStore + SenderKeyStore + SessionStoreExt + Sync + Clone,
655    >(
656        &mut self,
657        aci_protocol_store: &mut AciStore,
658        pni_protocol_store: &mut PniStore,
659        mut sender: MessageSender<AciOrPni, R>,
660        local_aci: Aci,
661        e164: PhoneNumber,
662        csprng: &mut R,
663    ) -> Result<(), MessageSenderError> {
664        let pni_identity_key_pair =
665            pni_protocol_store.get_identity_key_pair().await?;
666
667        let pni_identity_key = pni_identity_key_pair.identity_key();
668
669        // For every linked device, we generate a new set of pre-keys, and send them to the device.
670        let local_device_ids = aci_protocol_store
671            .get_sub_device_sessions(&local_aci.into())
672            .await?;
673
674        let mut device_messages =
675            Vec::<OutgoingPushMessage>::with_capacity(local_device_ids.len());
676        let mut device_pni_signed_prekeys =
677            HashMap::<String, SignedPreKeyEntity>::with_capacity(
678                local_device_ids.len(),
679            );
680        let mut device_pni_last_resort_kyber_prekeys =
681            HashMap::<String, KyberPreKeyEntity>::with_capacity(
682                local_device_ids.len(),
683            );
684        let mut pni_registration_ids =
685            HashMap::<String, u32>::with_capacity(local_device_ids.len());
686
687        let signature_valid_on_each_signed_pre_key = true;
688        for local_device_id in
689            std::iter::once(DEFAULT_DEVICE_ID).chain(local_device_ids)
690        {
691            let local_protocol_address =
692                local_aci.to_protocol_address(local_device_id);
693            let span = tracing::trace_span!(
694                "filtering devices",
695                address = %local_protocol_address
696            );
697            // Skip if we don't have a session with the device
698            if (local_device_id != DEFAULT_DEVICE_ID)
699                && aci_protocol_store
700                    .load_session(&local_protocol_address)
701                    .instrument(span)
702                    .await?
703                    .is_none()
704            {
705                tracing::warn!(
706                    "No session with device {}, skipping PNI provisioning",
707                    local_device_id
708                );
709                continue;
710            }
711            let (
712                _pre_keys,
713                signed_pre_key,
714                _kyber_pre_keys,
715                last_resort_kyber_prekey,
716            ) = if local_device_id == DEFAULT_DEVICE_ID {
717                crate::pre_keys::replenish_pre_keys(
718                    pni_protocol_store,
719                    csprng,
720                    &pni_identity_key_pair,
721                    true,
722                    0,
723                    0,
724                )
725                .await?
726            } else {
727                // Generate a signed prekey
728                let signed_pre_key_pair = KeyPair::generate(csprng);
729                let signed_pre_key_public = signed_pre_key_pair.public_key;
730                let signed_pre_key_signature =
731                    pni_identity_key_pair.private_key().calculate_signature(
732                        &signed_pre_key_public.serialize(),
733                        csprng,
734                    )?;
735
736                let signed_prekey_record = SignedPreKeyRecord::new(
737                    csprng.gen_range::<u32, _>(0..0xFFFFFF).into(),
738                    Timestamp::now(),
739                    &signed_pre_key_pair,
740                    &signed_pre_key_signature,
741                );
742
743                // Generate a last-resort Kyber prekey
744                let kyber_pre_key_record = KyberPreKeyRecord::generate(
745                    kem::KeyType::Kyber1024,
746                    csprng.gen_range::<u32, _>(0..0xFFFFFF).into(),
747                    pni_identity_key_pair.private_key(),
748                )?;
749                (
750                    vec![],
751                    signed_prekey_record,
752                    vec![],
753                    Some(kyber_pre_key_record),
754                )
755            };
756
757            let registration_id = if local_device_id == DEFAULT_DEVICE_ID {
758                pni_protocol_store.get_local_registration_id().await?
759            } else {
760                loop {
761                    let regid = generate_registration_id(csprng);
762                    if !pni_registration_ids.iter().any(|(_k, v)| *v == regid) {
763                        break regid;
764                    }
765                }
766            };
767
768            let local_device_id_s = local_device_id.to_string();
769            device_pni_signed_prekeys.insert(
770                local_device_id_s.clone(),
771                SignedPreKeyEntity::try_from(&signed_pre_key)?,
772            );
773            device_pni_last_resort_kyber_prekeys.insert(
774                local_device_id_s.clone(),
775                KyberPreKeyEntity::try_from(
776                    last_resort_kyber_prekey
777                        .as_ref()
778                        .expect("requested last resort key"),
779                )?,
780            );
781            pni_registration_ids
782                .insert(local_device_id_s.clone(), registration_id);
783
784            assert!(_pre_keys.is_empty());
785            assert!(_kyber_pre_keys.is_empty());
786
787            if local_device_id == DEFAULT_DEVICE_ID {
788                // This is the primary device
789                // We don't need to send a message to the primary device
790                continue;
791            }
792            // cfr. SignalServiceMessageSender::getEncryptedSyncPniInitializeDeviceMessage
793            let msg = SyncMessage {
794                pni_change_number: Some(PniChangeNumber {
795                    identity_key_pair: Some(
796                        pni_identity_key_pair.serialize().to_vec(),
797                    ),
798                    signed_pre_key: Some(signed_pre_key.serialize()?),
799                    last_resort_kyber_pre_key: Some(
800                        last_resort_kyber_prekey
801                            .expect("requested last resort key")
802                            .serialize()?,
803                    ),
804                    registration_id: Some(registration_id),
805                    new_e164: Some(
806                        e164.format().mode(phonenumber::Mode::E164).to_string(),
807                    ),
808                }),
809                padding: Some(random_length_padding(csprng, 512)),
810                ..SyncMessage::default()
811            };
812            let content: ContentBody = msg.into();
813            let msg = sender
814                .create_encrypted_message(
815                    &local_aci.into(),
816                    None,
817                    local_device_id.into(),
818                    &content.into_proto().encode_to_vec(),
819                )
820                .await?;
821            device_messages.push(msg);
822        }
823
824        self.service
825            .distribute_pni_keys(
826                pni_identity_key,
827                device_messages,
828                device_pni_signed_prekeys,
829                device_pni_last_resort_kyber_prekeys,
830                pni_registration_ids,
831                signature_valid_on_each_signed_pre_key,
832            )
833            .await?;
834
835        Ok(())
836    }
837}

839fn calculate_hmac256(
840    mac_key: &[u8],
841    ciphertext: &[u8],
842) -> Result<Output<Hmac<Sha256>>, ServiceError> {
843    let mut mac = Hmac::<Sha256>::new_from_slice(mac_key)
844        .map_err(|_| ServiceError::MacError)?;
845    mac.update(ciphertext);
846    Ok(mac.finalize().into_bytes())
847}

849pub fn encrypt_device_name<R: rand::Rng + rand::CryptoRng>(
850    csprng: &mut R,
851    device_name: &str,
852    identity_public: &IdentityKey,
853) -> Result<DeviceName, ServiceError> {
854    let plaintext = device_name.as_bytes().to_vec();
855    let ephemeral_key_pair = KeyPair::generate(csprng);
856
857    let master_secret = ephemeral_key_pair
858        .private_key
859        .calculate_agreement(identity_public.public_key())?;
860
861    let key1 = calculate_hmac256(&master_secret, b"auth")?;
862    let synthetic_iv = calculate_hmac256(&key1, &plaintext)?;
863    let synthetic_iv = &synthetic_iv[..16];
864
865    let key2 = calculate_hmac256(&master_secret, b"cipher")?;
866    let cipher_key = calculate_hmac256(&key2, synthetic_iv)?;
867
868    let mut ciphertext = plaintext;
869
870    const IV: [u8; 16] = [0; 16];
871    let mut cipher = Aes256Ctr128BE::new(&cipher_key, &IV.into());
872    cipher.apply_keystream(&mut ciphertext);
873
874    let device_name = DeviceName {
875        ephemeral_public: Some(
876            ephemeral_key_pair.public_key.serialize().to_vec(),
877        ),
878        synthetic_iv: Some(synthetic_iv.to_vec()),
879        ciphertext: Some(ciphertext),
880    };
881
882    Ok(device_name)
883}

885fn decrypt_device_name_from_device_info(
886    string: &str,
887    aci: &IdentityKeyPair,
888) -> Result<String, ServiceError> {
889    let data = BASE64_RELAXED.decode(string)?;
890    let name = DeviceName::decode(&*data)?;
891    crate::decrypt_device_name(aci.private_key(), &name)
892}
893
894pub fn decrypt_device_name(
895    private_key: &PrivateKey,
896    device_name: &DeviceName,
897) -> Result<String, ServiceError> {
898    let DeviceName {
899        ephemeral_public: Some(ephemeral_public),
900        synthetic_iv: Some(synthetic_iv),
901        ciphertext: Some(ciphertext),
902    } = device_name
903    else {
904        return Err(ServiceError::InvalidDeviceName);
905    };
906
907    let synthetic_iv: [u8; 16] = synthetic_iv[..synthetic_iv.len().min(16)]
908        .try_into()
909        .map_err(|_| ServiceError::MacError)?;
910
911    let ephemeral_public = PublicKey::deserialize(ephemeral_public)?;
912
913    let master_secret = private_key.calculate_agreement(&ephemeral_public)?;
914    let key2 = calculate_hmac256(&master_secret, b"cipher")?;
915    let cipher_key = calculate_hmac256(&key2, &synthetic_iv)?;
916
917    let mut plaintext = ciphertext.to_vec();
918    const IV: [u8; 16] = [0; 16];
919    let mut cipher =
920        Aes256Ctr128BE::new(cipher_key.as_slice().into(), &IV.into());
921    cipher.apply_keystream(&mut plaintext);
922
923    let key1 = calculate_hmac256(&master_secret, b"auth")?;
924    let our_synthetic_iv = calculate_hmac256(&key1, &plaintext)?;
925    let our_synthetic_iv = &our_synthetic_iv[..16];
926
927    if synthetic_iv != our_synthetic_iv {
928        Err(ServiceError::MacError)
929    } else {
930        Ok(String::from_utf8_lossy(&plaintext).to_string())
931    }
932}
