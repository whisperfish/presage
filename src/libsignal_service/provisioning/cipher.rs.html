20enum CipherMode {
21    DecryptAndEncrypt(KeyPair),
22    EncryptOnly(PublicKey),
23}
24
25impl Debug for CipherMode {
26    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
27        match self {
28            CipherMode::DecryptAndEncrypt(key_pair) => f
29                .debug_tuple("CipherMode::DecryptAndEncrypt")
30                .field(&key_pair.public_key)
31                .finish(),
32            CipherMode::EncryptOnly(public) => f
33                .debug_tuple("CipherMode::EncryptOnly")
34                .field(&public)
35                .finish(),
36        }
37    }
38}
39
40impl CipherMode {
41    fn public(&self) -> &PublicKey {
42        match self {
43            CipherMode::DecryptAndEncrypt(pair) => &pair.public_key,
44            CipherMode::EncryptOnly(pub_key) => pub_key,
45        }
46    }
47}

49const VERSION: u8 = 1;
50
51#[derive(Debug)]
52pub struct ProvisioningCipher {
53    key_material: CipherMode,
54}
55
56impl ProvisioningCipher {
57    pub fn from_public(key: PublicKey) -> Self {
58        Self {
59            key_material: CipherMode::EncryptOnly(key),
60        }
61    }
62
63    pub fn from_key_pair(key_pair: KeyPair) -> Self {
64        Self {
65            key_material: CipherMode::DecryptAndEncrypt(key_pair),
66        }
67    }
68
69    pub fn public_key(&self) -> &PublicKey {
70        self.key_material.public()
71    }
72
73    pub fn encrypt<R: Rng + CryptoRng>(
74        &self,
75        csprng: &mut R,
76        msg: ProvisionMessage,
77    ) -> Result<ProvisionEnvelope, ProvisioningError> {
78        let msg = msg.encode_to_vec();
79
80        let our_key_pair = libsignal_protocol::KeyPair::generate(csprng);
81        let agreement = our_key_pair.calculate_agreement(self.public_key())?;
82
83        let mut shared_secrets = [0; 64];
84        hkdf::Hkdf::<sha2::Sha256>::new(None, &agreement)
85            .expand(b"TextSecure Provisioning Message", &mut shared_secrets)
86            .expect("valid output length");
87
88        let aes_key = &shared_secrets[0..32];
89        let mac_key = &shared_secrets[32..];
90        let iv: [u8; IV_LENGTH] = csprng.gen();
91
92        let cipher = cbc::Encryptor::<Aes256>::new(aes_key.into(), &iv.into());
93        let ciphertext = cipher.encrypt_padded_vec_mut::<Pkcs7>(&msg);
94        let mut mac = Hmac::<Sha256>::new_from_slice(mac_key)
95            .expect("HMAC can take any size key");
96        mac.update(&[VERSION]);
97        mac.update(&iv);
98        mac.update(&ciphertext);
99        let mac = mac.finalize().into_bytes();
100
101        let body: Vec<u8> = std::iter::once(VERSION)
102            .chain(iv.iter().cloned())
103            .chain(ciphertext)
104            .chain(mac)
105            .collect();
106
107        Ok(ProvisionEnvelope {
108            public_key: Some(our_key_pair.public_key.serialize().into()),
109            body: Some(body),
110        })
111    }
112
113    pub fn decrypt(
114        &self,
115        provision_envelope: ProvisionEnvelope,
116    ) -> Result<ProvisionMessage, ProvisioningError> {
117        let key_pair = match self.key_material {
118            CipherMode::DecryptAndEncrypt(ref key_pair) => key_pair,
119            CipherMode::EncryptOnly(_) => {
120                return Err(ProvisioningError::EncryptOnlyProvisioningCipher);
121            },
122        };
123        let master_ephemeral = PublicKey::deserialize(
124            &provision_envelope.public_key.expect("no public key"),
125        )?;
126        let body = provision_envelope
127            .body
128            .expect("no body in ProvisionMessage");
129        if body[0] != VERSION {
130            return Err(ProvisioningError::BadVersionNumber);
131        }
132
133        let iv = &body[IV_OFFSET..(IV_LENGTH + IV_OFFSET)];
134        let mac = &body[(body.len() - 32)..];
135        let cipher_text = &body[16 + 1..(body.len() - CIPHER_KEY_SIZE)];
136        let iv_and_cipher_text = &body[0..(body.len() - CIPHER_KEY_SIZE)];
137        debug_assert_eq!(iv.len(), IV_LENGTH);
138        debug_assert_eq!(mac.len(), 32);
139
140        let agreement = key_pair.calculate_agreement(&master_ephemeral)?;
141
142        let mut shared_secrets = [0; 64];
143        hkdf::Hkdf::<sha2::Sha256>::new(None, &agreement)
144            .expand(b"TextSecure Provisioning Message", &mut shared_secrets)
145            .expect("valid output length");
146
147        let parts1 = &shared_secrets[0..32];
148        let parts2 = &shared_secrets[32..];
149
150        let mut verifier = Hmac::<Sha256>::new_from_slice(parts2)
151            .expect("HMAC can take any size key");
152        verifier.update(iv_and_cipher_text);
153        let our_mac = verifier.finalize().into_bytes();
154        debug_assert_eq!(our_mac.len(), mac.len());
155        if &our_mac[..32] != mac {
156            return Err(ProvisioningError::MismatchedMac);
157        }
158
159        // libsignal-service-java uses Pkcs5,
160        // but that should not matter.
161        // https://crypto.stackexchange.com/questions/9043/what-is-the-difference-between-pkcs5-padding-and-pkcs7-padding
162        let cipher = cbc::Decryptor::<Aes256>::new(parts1.into(), iv.into());
163        let input = cipher
164            .decrypt_padded_vec_mut::<Pkcs7>(cipher_text)
165            .map_err(ProvisioningError::AesPaddingError)?;
166
167        Ok(prost::Message::decode(Bytes::from(input))?)
168    }

171#[cfg(test)]
172mod tests {
173    use super::*;
174
175    #[test]
176    fn encrypt_provisioning_roundtrip() -> anyhow::Result<()> {
177        let mut rng = rand::thread_rng();
178        let key_pair = KeyPair::generate(&mut rng);
179        let cipher = ProvisioningCipher::from_key_pair(key_pair);
180        let encrypt_cipher: ProvisioningCipher =
181            ProvisioningCipher::from_public(*cipher.public_key());
182
183        assert_eq!(
184            cipher.public_key(),
185            encrypt_cipher.public_key(),
186            "copy public key"
187        );
188
189        let msg = ProvisionMessage::default();
190        let encrypted = encrypt_cipher.encrypt(&mut rng, msg.clone())?;
191
192        assert!(matches!(
193            encrypt_cipher.decrypt(encrypted.clone()),
194            Err(ProvisioningError::EncryptOnlyProvisioningCipher)
195        ));
196
197        let decrypted = cipher.decrypt(encrypted)?;
198        assert_eq!(msg, decrypted);
199
200        Ok(())
201    }
