12pub(crate) struct MessageKeys {
13    cipher_key: [u8; 32],
14    mac_key: [u8; 32],
15    iv: [u8; 16],
16    counter: u32,
17}
18
19impl MessageKeys {
20    pub(crate) fn derive_keys(input_key_material: &[u8], counter: u32) -> Self {
21        let mut okm = [0; 80];
22        hkdf::Hkdf::<sha2::Sha256>::new(None, input_key_material)
23            .expand(b"WhisperMessageKeys", &mut okm)
24            .expect("valid output length");
25
26        MessageKeys {
27            cipher_key: *array_ref![okm, 0, 32],
28            mac_key: *array_ref![okm, 32, 32],
29            iv: *array_ref![okm, 64, 16],
30            counter,
31        }
32    }
33
34    pub(crate) fn new(cipher_key: [u8; 32], mac_key: [u8; 32], iv: [u8; 16], counter: u32) -> Self {
35        MessageKeys {
36            cipher_key,
37            mac_key,
38            iv,
39            counter,
40        }
41    }
42
43    #[inline]
44    pub(crate) fn cipher_key(&self) -> &[u8; 32] {
45        &self.cipher_key
46    }
47
48    #[inline]
49    pub(crate) fn mac_key(&self) -> &[u8; 32] {
50        &self.mac_key
51    }
52
53    #[inline]
54    pub(crate) fn iv(&self) -> &[u8; 16] {
55        &self.iv
56    }
57
58    #[inline]
59    pub(crate) fn counter(&self) -> u32 {
60        self.counter
61    }
62}

64#[derive(Clone, Debug)]
65pub(crate) struct ChainKey {
66    key: [u8; 32],
67    index: u32,
68}
69
70impl ChainKey {
71    const MESSAGE_KEY_SEED: [u8; 1] = [0x01u8];
72    const CHAIN_KEY_SEED: [u8; 1] = [0x02u8];
73
74    pub(crate) fn new(key: [u8; 32], index: u32) -> Self {
75        Self { key, index }
76    }
77
78    #[inline]
79    pub(crate) fn key(&self) -> &[u8; 32] {
80        &self.key
81    }
82
83    #[inline]
84    pub(crate) fn index(&self) -> u32 {
85        self.index
86    }
87
88    pub(crate) fn next_chain_key(&self) -> Self {
89        Self {
90            key: self.calculate_base_material(Self::CHAIN_KEY_SEED),
91            index: self.index + 1,
92        }
93    }
94
95    pub(crate) fn message_keys(&self) -> MessageKeys {
96        MessageKeys::derive_keys(
97            &self.calculate_base_material(Self::MESSAGE_KEY_SEED),
98            self.index,
99        )
100    }
101
102    fn calculate_base_material(&self, seed: [u8; 1]) -> [u8; 32] {
103        crypto::hmac_sha256(&self.key, &seed)
104    }
105}

107#[derive(Clone, Debug)]
108pub(crate) struct RootKey {
109    key: [u8; 32],
110}
111
112impl RootKey {
113    pub(crate) fn new(key: [u8; 32]) -> Self {
114        Self { key }
115    }
116
117    pub(crate) fn key(&self) -> &[u8; 32] {
118        &self.key
119    }
120
121    pub(crate) fn create_chain(
122        self,
123        their_ratchet_key: &PublicKey,
124        our_ratchet_key: &PrivateKey,
125    ) -> Result<(RootKey, ChainKey)> {
126        let shared_secret = our_ratchet_key.calculate_agreement(their_ratchet_key)?;
127        let mut derived_secret_bytes = [0; 64];
128        hkdf::Hkdf::<sha2::Sha256>::new(Some(&self.key), &shared_secret)
129            .expand(b"WhisperRatchet", &mut derived_secret_bytes)
130            .expect("valid output length");
131
132        Ok((
133            RootKey {
134                key: *array_ref![derived_secret_bytes, 0, 32],
135            },
136            ChainKey {
137                key: *array_ref![derived_secret_bytes, 32, 32],
138                index: 0,
139            },
140        ))
141    }
142}
143
144impl fmt::Display for RootKey {
145    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
146        write!(f, "{}", hex::encode(self.key))
147    }
148}

156        let seed = [
157            0x8au8, 0xb7, 0x2d, 0x6f, 0x4c, 0xc5, 0xac, 0x0d, 0x38, 0x7e, 0xaf, 0x46, 0x33, 0x78,

159            0x2e, 0x7a, 0xd4, 0x8f,
160        ];
161        let message_key = [
162            0xbfu8, 0x51, 0xe9, 0xd7, 0x5e, 0x0e, 0x31, 0x03, 0x10, 0x51, 0xf8, 0x2a, 0x24, 0x91,
163            0xff, 0xc0, 0x84, 0xfa, 0x29, 0x8b, 0x77, 0x93, 0xbd, 0x9d, 0xb6, 0x20, 0x05, 0x6f,
164            0xeb, 0xf4, 0x52, 0x17,
165        ];
166        let mac_key = [
167            0xc6u8, 0xc7, 0x7d, 0x6a, 0x73, 0xa3, 0x54, 0x33, 0x7a, 0x56, 0x43, 0x5e, 0x34, 0x60,
168            0x7d, 0xfe, 0x48, 0xe3, 0xac, 0xe1, 0x4e, 0x77, 0x31, 0x4d, 0xc6, 0xab, 0xc1, 0x72,
169            0xe7, 0xa7, 0x03, 0x0b,

171        let next_chain_key = [
172            0x28u8, 0xe8, 0xf8, 0xfe, 0xe5, 0x4b, 0x80, 0x1e, 0xef, 0x7c, 0x5c, 0xfb, 0x2f, 0x17,

185        assert_eq!(1, chain_key.next_chain_key().message_keys().counter());
186        Ok(())
