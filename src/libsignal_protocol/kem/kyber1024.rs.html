6use libcrux_ml_kem::mlkem1024::{MlKem1024Ciphertext, MlKem1024PrivateKey, MlKem1024PublicKey};
7use libcrux_ml_kem::{SHARED_SECRET_SIZE, kyber1024};
8use rand::{CryptoRng, Rng as _};
9
10use super::{
11    BadKEMKeyLength, ConstantLength as _, DecapsulateError, KeyMaterial, KeyType, Public, Secret,
12};
13
14pub(crate) struct Parameters;
15
16impl super::Parameters for Parameters {
17    const KEY_TYPE: KeyType = KeyType::Kyber1024;
18    const PUBLIC_KEY_LENGTH: usize = MlKem1024PublicKey::LENGTH;
19    const SECRET_KEY_LENGTH: usize = MlKem1024PrivateKey::LENGTH;
20    const CIPHERTEXT_LENGTH: usize = MlKem1024Ciphertext::LENGTH;
21    const SHARED_SECRET_LENGTH: usize = SHARED_SECRET_SIZE;
22
23    fn generate<R: CryptoRng + ?Sized>(
24        csprng: &mut R,
25    ) -> (KeyMaterial<Public>, KeyMaterial<Secret>) {
26        let (sk, pk) = kyber1024::generate_key_pair(csprng.random()).into_parts();
27        (KeyMaterial::from(pk), KeyMaterial::from(sk))
28    }
29
30    fn encapsulate<R: CryptoRng + ?Sized>(
31        pub_key: &KeyMaterial<Public>,
32        csprng: &mut R,
33    ) -> Result<(Box<[u8]>, Box<[u8]>), BadKEMKeyLength> {
34        let kyber_pk =
35            MlKem1024PublicKey::try_from(pub_key.as_ref()).map_err(|_| BadKEMKeyLength)?;
36        let (kyber_ct, kyber_ss) = kyber1024::encapsulate(&kyber_pk, csprng.random());
37        Ok((kyber_ss.as_ref().into(), kyber_ct.as_ref().into()))
38    }
39
40    fn decapsulate(
41        secret_key: &KeyMaterial<Secret>,
42        ciphertext: &[u8],
43    ) -> Result<Box<[u8]>, DecapsulateError> {
44        let kyber_sk = MlKem1024PrivateKey::try_from(secret_key.as_ref())
45            .map_err(|_| DecapsulateError::BadKeyLength)?;
46        let kyber_ct = MlKem1024Ciphertext::try_from(ciphertext)
47            .map_err(|_| DecapsulateError::BadCiphertext)?;
48        let kyber_ss = kyber1024::decapsulate(&kyber_sk, &kyber_ct);
49
50        Ok(kyber_ss.as_ref().into())
51    }
52}
