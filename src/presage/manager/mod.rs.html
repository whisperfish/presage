15/// Signal manager
16///
17/// The manager is parametrized over the [`crate::store::Store`] which stores the configuration, keys and
18/// optionally messages; and over the type state which describes what is the current state of the
19/// manager: in registration, linking, TODO
20///
21/// Depending on the state specific methods are available or not.
22pub struct Manager<Store, State> {
23    /// Implementation of a metadata and messages store
24    store: Store,
25    /// Part of the manager which is persisted in the store.
26    state: Arc<State>,
27}

36impl<Store: Clone, State> Clone for Manager<Store, State> {
37    fn clone(&self) -> Self {
38        Self {
39            store: self.store.clone(),
40            state: self.state.clone(),
41        }
42    }
43}
44
45impl<Store, State: fmt::Debug> fmt::Debug for Manager<Store, State> {
46    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
47        f.debug_struct("Manager")
48            .field("state", &self.state)
49            .finish_non_exhaustive()
50    }
51}
