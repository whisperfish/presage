9/// The error type of Signal manager
10#[derive(thiserror::Error, Debug)]
11#[non_exhaustive]
12pub enum Error<S: std::error::Error> {
13    #[error("captcha from https://signalcaptchas.org/registration/generate.html required")]
14    CaptchaRequired,
15    #[error("input/output error: {0}")]
16    IoError(#[from] std::io::Error),
17    #[error("JSON error: {0}")]
18    JsonError(#[from] serde_json::Error),
19    #[error("error decoding base64 data: {0}")]
20    Base64Error(#[from] base64::DecodeError),
21    #[error("wrong slice size: {0}")]
22    TryFromSliceError(#[from] std::array::TryFromSliceError),
23    #[error("phone number parsing error: {0}")]
24    PhoneNumberError(#[from] libsignal_service::prelude::phonenumber::ParseError),
25    #[error("UUID decoding error: {0}")]
26    UuidError(#[from] libsignal_service::prelude::UuidError),
27    #[error("libsignal-protocol error: {0}")]
28    ProtocolError(#[from] SignalProtocolError),
29    #[error("libsignal-service error: {0}")]
30    ServiceError(#[from] libsignal_service::prelude::ServiceError),
31    #[error("libsignal-service error: {0}")]
32    ProfileManagerError(#[from] libsignal_service::ProfileManagerError),
33    #[error("libsignal-service sending error: {0}")]
34    MessageSenderError(Box<MessageSenderError>),
35    #[error("this client is already registered with Signal")]
36    AlreadyRegisteredError,
37    #[error("this client is not yet registered, please register or link as a secondary device")]
38    NotYetRegisteredError,
39    #[error("failed to provision device: {0}")]
40    ProvisioningError(#[from] libsignal_service::provisioning::ProvisioningError),
41    #[error("no provisioning message received")]
42    NoProvisioningMessageReceived,
43    #[error("qr code error")]
44    LinkingError,
45    #[error("please relink your client")]
46    RelinkNecessary,
47    #[error("missing key {0} in config DB")]
48    MissingKeyError(Cow<'static, str>),
49    #[error("message pipe not started, you need to start receiving messages before you can send anything back")]
50    MessagePipeNotStarted,
51    #[error("receiving pipe was interrupted")]
52    MessagePipeInterruptedError,
53    #[error("failed to parse contact information: {0}")]
54    ParseContactError(#[from] ParseContactError),
55    #[error("failed to decrypt attachment: {0}")]
56    AttachmentCipherError(#[from] libsignal_service::attachment_cipher::AttachmentCipherError),
57    #[error("unknown group")]
58    UnknownGroup,
59    #[error("unknown recipient")]
60    UnknownRecipient,
61    #[error("timeout: {0}")]
62    Timeout(#[from] tokio::time::error::Elapsed),
63    #[error("store error: {0}")]
64    Store(S),
65    #[error("push challenge required (not implemented)")]
66    PushChallengeRequired,
67    #[error("Not allowed to request verification code, reason unknown: {0:?}")]
68    RequestingCodeForbidden(RegistrationSessionMetadataResponse),
69    #[error("attachment sha256 checksum did not match")]
70    UnexpectedAttachmentChecksum,
71    #[error("Unverified registration session (i.e. wrong verification code)")]
72    UnverifiedRegistrationSession,
73    #[error("profile cipher error")]
74    ProfileCipherError(#[from] libsignal_service::profile_cipher::ProfileCipherError),
75    #[error("An operation was requested that requires the registration to be primary, but it was only secondary")]
76    NotPrimaryDevice,
77    #[error("Failed to get initial messages after uploading pre-keys")]
78    UpdatePreKeyFailure,
79    #[error("invalid device ID (out of bounds)")]
80    InvalidDeviceId,
81}

83impl<S: std::error::Error> From<MessageSenderError> for Error<S> {
